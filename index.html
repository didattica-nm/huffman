 18void AsyncFSM::handleSerialEvt(SerialEvent *serialEvt)
 19{
 20    Msg *receivedMsg = serialEvt->getValue();
 21    MsgTopic msgTopic = receivedMsg->getTopic();
 22    switch (msgTopic)
 23    {
 24    case MsgTopic::MODE: 
 25    {
 26        // new mode received from CUS,
 27        // decoding which mode was received
 28        if (receivedMsg->getContent().equals("MANUAL"))
 29        {
 30            this->state->setState(SystemState::MANUAL);
 31            // nothing else to do
 32
 33        } else if (receivedMsg->getContent().equals("UNCONNECTED"))
 34        {
 35            this->state->setState(SystemState::UNCONNECTED);
 36            // nothing else to do   
 37        } 
 38        else if (receivedMsg->getContent().equals("AUTOMATIC"))
 39        {
 40            // AUTOMATIC mode received
 41            this->state->setState(SystemState::AUTOMATIC);
 42        } 
 43        else 
 44        {
 45            #ifdef DEBUG
 46            log("unexpected mode received!");
 47            #endif
 48        }
 49        displayState();
 50        break;
 51    }
 52    case MsgTopic::VALVE: 
 53    {
 54        // we received something about the valve
 55        // assuming the opening value arrives already in percentage between [0.0, 1.0]
 56        this->valve->setOpening(receivedMsg->getContent().toFloat());
 57        break;
 58    }
 59    default: 
 60    {
 61        #ifdef DEBUG
 62        log("Unexpected msg topic received!");
 63        #endif
 64        break;
 65    }
 66    }
 67}

 69void AsyncFSM::handleButtonEvt(ButtonEvent *buttonEvt)
 70{
 71    SystemState currentSystemState = this->state->getState();
 72    SystemState newState = currentSystemState;
 73    switch (currentSystemState)
 74    {
 75        case SystemState::UNCONNECTED:
 76        {
 77            /* do nothing */
 78            break;
 79        }
 80        case SystemState::AUTOMATIC:
 81        {
 82            newState = SystemState::MANUAL;
 83            break;
 84        }
 85        case SystemState::MANUAL:
 86        {
 87            newState = SystemState::AUTOMATIC;
 88            break;
 89        }
 90    }
 91
 92    this->state->setState(newState);
 93    String newStateStr = "UNCONNECTED";
 94    if (newState == SystemState::AUTOMATIC) { newStateStr = "AUTOMATIC"; }
 95    else if (newState == SystemState::MANUAL) { newStateStr = "MANUAL"; }
 96    msgService.sendMsg("MODE:" + newStateStr);
 97    displayState();
 98}

100void AsyncFSM::handlePotEvt(PotEvent *potEvt)
101{
102    SystemState currentSystemState = this->state->getState();
103    if (currentSystemState != SystemState::MANUAL) { return; }
104
105    float percentage = (float)(potEvt->getValue() - POT_MIN) / (POT_MAX - POT_MIN);
106    valve->setOpening(percentage);
107    msgService.sendMsg("VALVE:" + String(percentage));
108}

110void AsyncFSM::checkAndProcessEvent()
111{
112    noInterrupts();
113    bool isEmpty = sharedQueue.isEmpty();
114    interrupts();
115
116    // Exit early if no events occurred
117    if (isEmpty) { return; }
118    
119    noInterrupts();
120    IEvent *evt = sharedQueue.dequeue();
121    interrupts();
122
123    switch (evt->getType())
124    {
125    case EventType::SERIAL_EVT: 
126    {
127        SerialEvent *serialEvt = static_cast<SerialEvent *>(evt);
128        #ifdef DEBUG
129        log("serial event!");
130        #endif
131        handleSerialEvt(serialEvt);
132        delete serialEvt;
133        break;
134    }
135    case EventType::BUTTON_EVT:
136    {
137        ButtonEvent *buttonEvt = static_cast<ButtonEvent *>(evt);
138        #ifdef DEBUG
139        log("button event!");
140        #endif
141        handleButtonEvt(buttonEvt);
142        delete buttonEvt;
143        break;
144    }
145    case EventType::POT_EVT: 
146    {
147        PotEvent *potEvt = static_cast<PotEvent *>(evt);
148        #ifdef DEBUG
149        log("pot event!");
150        #endif
151        handlePotEvt(potEvt);
152        delete potEvt;
153        break;
154    }
155    default:
156        #ifdef DEBUG
157        log("unexpected event received!");
158        #endif
159        break;
160    }
161}

163void AsyncFSM::displayState()
164{
165    String msg = "UNCONNECTED";
166    switch (state->getState())
167    {
168        case SystemState::AUTOMATIC: { msg = "AUTOMATIC"; break; }
169        case SystemState::MANUAL: { msg = "MANUAL"; break; }
170        default: break;
171    }
172    lcd->print(msg);
173    #ifdef DEBUG
174    log(msg);
175    #endif
176}

$\textbf{Exercise}$

Prove that $a_n = \frac{2n + 1}{n + 2} \xrightarrow[n \to +\infty]{} 2$.

$\textbf{Proof}$

Given $\varepsilon > 0,\ \exists m_\varepsilon \in \mathbb{N}:$

\begin{equation} \begin{split} \left\lvert a_n - l \right\rvert &= \left\lvert \frac{2n + 1}{n + 2} - 2 \right\rvert \leq \varepsilon, \qquad \forall n \in \mathbb{N} \\ &= \left\lvert \frac{2n + 1 -2n - 4}{n + 2} \right\rvert \leq \varepsilon \\ &\Longleftrightarrow \frac{3}{n + 2} \leq \varepsilon \\ &\Longleftrightarrow n + 2 \geq \frac{3}{\varepsilon} \\ &\Longleftrightarrow n \geq \frac{3}{\varepsilon} - 2 \end{split} \end{equation}

I can choose $m_\varepsilon = \left[\frac{3}{\varepsilon} - 2\right] + 1$ in order to satisfy the limit. $\#$
